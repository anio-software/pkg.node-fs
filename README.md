# @anio-software/fs

Collection of async/sync fs methods.

```js
import {
	clean,
	cleanSync,

	hashFile,
	hashFileSync,

	isDirectory,
	isDirectorySync,

	isFile,
	isFileSync,

	isRegularDirectory,
	isRegularDirectorySync,

	isRegularFile,
	isRegularFileSync,

	mkdirp,
	mkdirpSync,

	readFileInChunks,
	readFileInChunksSync,

	remove,
	removeSync,

	scandir,
	scandirSync
} from "@anio-software/fs"
```
